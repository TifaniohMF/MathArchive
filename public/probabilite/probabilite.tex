\section{Probabilitﾃｩ}
$\mathcal{F} = P(\Omega), \; \mathcal{F}$ est ici une partie de $\Omega$ oﾃｹ $\Omega$ est l'ensemble de tous les rﾃｩsultats possibles. $\mathcal{F}$ est appelﾃｩ un tribus et ($\mathcal{F}, \Omega$) un espace mesurable.

\begin{defin}
	Une mesure de probabilitﾃｩ (ou probabilitﾃｩ) est une application P de $\mathcal{F}$ vers [0, 1] qui verifie les conditions suivantes:
	\begin{enumerate}
		\item P($\Omega$) = 1 et P($\emptyset$) = 0,
		\item Soit $(A_{i})_{i \in \mathbb{N}}$ un collection de famille d'ﾃｩvenement finie ou dﾃｩnombrable. Si les $A_{i}$ sont deux ﾃ deux disjoints, on a $P(\bigcup_{i \in  \mathbb{N}}A_{i}) = \sum_{i \in \mathbb{N}} P(A_{i})$ ($\sigma$-additivitﾃｩ)
	\end{enumerate}
	Le triplet $(\Omega, \mathcal{F}, P)$ est un espace de probabilitﾃｩ. On dit qu'un probabilitﾃｩ d'un ﾃｩvenement A est presque sﾃｻr si $P(A) = 1$, il est nﾃｩgligeable si $P(A)=0$.
\end{defin} 

\begin{prop}
	\begin{enumerate}
		\item $P(A^{c}) = 1 - P(A)$,
		\item $P(A \cup B) = P(A) + P(B) - P(A \cap B)$,
		\item Si $A \subset B$, alors $P(A) \leq P(B)$,
		\item Soit $(A_{i})_{i \in \mathbb{N}}$ une famille d'ﾃｩvenement finie ou dﾃｩnombrable. On a\\ $P(B) = \sum_{i \in \mathbb{N}} P(A_{i} \cap B)$ (\textbf{Formule de dﾃｩcomposition})
	\end{enumerate}
\end{prop}
\textit{Preuve.}
\begin{enumerate}
	\item Montrons que $P(A^{c}) = 1 - P(A)$,\\
	On sait que $A \cup A^{c} = \Omega$, Comme $A$ et $A^{c}$ sont disjoints. D'aprﾃｨs $\sigma$-additivitﾃｩ, on a $P(A \cup A^{c}) = P(A) + P(A^{c})$. Or $P(A \cup A^{c})=P(\Omega)=1$. Donc $P(A^{c}) = 1 - P(A)$.
	\item Montrons que $P(A \cup B) = P(A) + P(B) - P(A \cap B)$,\\
		On peut dﾃｩcomposer $A \cup B$ comme suit, $A \cup B = A \cup (B \cap A^{c})$. A et $B \cap A^{c}$ sont disjoints. D'aprﾃｨs $\sigma$-additivitﾃｩ, on a $P(A \cup (B \cap A^{c})) = P(A) + P(B \cap A^{c})$. En dﾃｩcomposant aussi B, on a $B = (B \cap A) \cup (B \cup A^{c})$. Par suit, on a $P(B) = P(B \cap A) \cup (B \cup A^{c}) = P(B \cap A) + P(B \cup A^{c})$. On a alors $P(B \cup A^{c}) = P(B) - P(A \cap B)$. D'oﾃｹ $P(A \cup B) = P(A) + P(B) - P(A \cap B)$.
\end{enumerate}
